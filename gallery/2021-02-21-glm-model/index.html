Data

dat <- attitude
dat$high.rating <- (dat$rating > 70)

Model

probit.model <- glm(high.rating ~ learning + critical +
   advance, data=dat, family = binomial(link = "probit"))

flextable

ft <- as_flextable(probit.model)
